DECLARE @PATH VARCHAR(100)

SELECT @PATH = 'C:\FOTOS\'

 SELECT IMAGEM,
                ((@PATH+CPF)+'.jpg') AS ENDERECO
FROM VW_RM_EFIX_FOTOS

 
CREATE VIEW VW_RM_EFIX_FOTOS
AS

SELECT      GIMAGEM.IMAGEM, PPESSOA.CPF
FROM  PPESSOA
                 INNER JOIN GIMAGEM
                       ON GIMAGEM.ID = PPESSOA.IDIMAGEM
